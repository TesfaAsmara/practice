            n = n/2
            sequence.append(n)
        else:
            n = (n*3)+1

def minesweeper2(file):

    def count_mines(arr):
        num_x = len(arr)
        num_y = len(arr[0])
        field = [[0 for i in range(num_y)] for i in range(num_x)]
        for i in range(num_x):
            for j in range(num_y):
                position = arr[i][j]
                if position == "*":
                    field[i][j] = '*'
                    for row in [i+1,i, i-1]:
                        if row >= 0 and row < num_x:
                            for col in [j+1,j, j-1]:
                                if col >= 0 and col < num_y:
                                        if field[row][col] != "*":
                                            field[row][col] += 1
        output = ""
        for i, row in enumerate(field):
            output += "".join(str(x) for x in row) + "\n"
        return output
    
    input = [line.strip() for line in fin.readlines()]
    file_index = 0
    output = ''

            break

def equalize_cost(file):
    fin = open(file, "r") 
    fout = open(file.replace(".in", ".out"), "w")

    input = [float(f"{float(line.strip()):2f}") for line in fin.readlines()]

    def read(expenses):
        #str(max(abs(x) for x in [x - y if i != j else -1 for i, x in enumerate(expenses) for j, y in enumerate(expenses)]))
        return sum([x if x > 0 else 0 for x in list(map(lambda x: x - float(f"{sum(expenses)/len(expenses):2f}"), expenses))])
    
    file_index = 0
    output = ""
    while file_index < len(input):
        n = int(input[file_index])
        if n != 0:
            start = file_index+1
            stop = file_index+n+1
            output += f"{read(input[start:stop]):2f}\n"
            file_index = stop
        else:
            break
    fout.write(output)

    def find_region_r(x, y, image, region):
        rows = len(image)
        cols = len(image[0])
        cur_pix = image[x][y]
        n_pix = image[x][y+1] if y+1 < cols else ""
        n_cor = [x, y+1] if y+1 < cols else []
        e_pix = image[x+1][y] if x+1 < rows else ""
        e_cor = [x+1, y] if x+1 < rows else []
        s_pix = image[x][y-1] if y-1 > 0 else ""
        s_cor = [x, y-1] if y-1 > 0 else [] 
        w_pix = image[x-1][y] if x-1 > 0 else ""
        w_cor = [x-1, y] if x-1 > 0 else []
        if cur_pix == n_pix and n_cor not in region:
            region.append(n_cor)
            region.append(find_region_r(n_cor[0], n_cor[1], image, region))
        elif cur_pix == e_pix and e_cor not in region:
            region.append(e_cor)
            region.append(find_region_r(e_cor[0], e_cor[1], image, region))

        elif cur_pix == w_pix and w_cor not in region:
            region.append(w_cor)
            region.append(find_region_r(w_cor[0], w_cor[1], image, region))
        return region

def compute_nesw(cors):
    rows = cors[0]
    cols = cors[1]
    return [(rows, cols+1), (rows-1, cols), (rows, cols-1), (rows+1, cols)]
